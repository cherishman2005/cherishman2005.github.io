# C++ set/map模拟最小堆，最大堆

```
    vector<int> nums = {1,2,2,5,3,5};

    set<int, less<int>> s1;
    s1.insert(nums.begin(), nums.end());
    printf("\n");
    for (const auto & e : s1)
    {
        printf("->%d", e);
    }
    printf("\n");


    set<int, greater<int>> s2;
    s2.insert(nums.begin(), nums.end());
    printf("\n");
    for (const auto & e : s2)
    {
        printf("->%d", e);
    }
    printf("\n");
```
